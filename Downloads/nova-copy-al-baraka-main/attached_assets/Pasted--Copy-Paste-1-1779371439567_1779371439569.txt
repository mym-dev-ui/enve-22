B) Presence (متصل/غير متصل) عبر RTDB

لازم يكون عندك RTDB مفعّل.
lib/presence.ts

import { getDatabase, ref, onValue, set, serverTimestamp, onDisconnect } from "firebase/database";
import { app } from "./firebase"; // عدّلها حسب ملف firebase عندك
export function startPresence(uid: string) {
  const db = getDatabase(app);
  const statusRef = ref(db, `/status/${uid}`);
  const connectedRef = ref(db, ".info/connected");
  onValue(connectedRef, (snap) => {
    if (snap.val() === false) return;
    // لما يقطع اتصال: يصير offline
    onDisconnect(statusRef).set({
      state: "offline",
      lastChanged: serverTimestamp(),
    });
    // لما يكون متصل
    set(statusRef, {
      state: "online",
      lastChanged: serverTimestamp(),
    });
  });
}
export function listenPresence(uid: string, cb: (state: "online" | "offline") => void) {
  const db = getDatabase(app);
  const statusRef = ref(db, `/status/${uid}`);
  return onValue(statusRef, (snap) => {
    const v = snap.val();
    cb(v?.state === "online" ? "online" : "offline");
  });
}

C) Realtime Firestore للزوار + الأكواد (onSnapshot سريع)

lib/realtime.ts

import { collection, doc, onSnapshot, orderBy, query, limit } from "firebase/firestore";
import { db } from "./firebase"; // عدّلها حسب ملف firebase عندك
export function listenVisitors(cb: (rows: any[]) => void) {
  const q = query(collection(db, "visitors"), orderBy("createdAt", "desc"), limit(50));
  return onSnapshot(q, (snap) => {
    cb(snap.docs.map((d) => ({ id: d.id, ...d.data() })));
  });
}
// codes داخل visitors/{id}/codes
export function listenVisitorCodes(visitorId: string, cb: (codes: any[]) => void) {
  const q = query(
    collection(db, "visitors", visitorId, "codes"),
    orderBy("createdAt", "desc"),
    limit(20)
  );
  return onSnapshot(q, (snap) => {
    cb(snap.docs.map((d) => ({ id: d.id, ...d.data() })));
  });
}
export function listenVisitorProfile(visitorId: string, cb: (data: any) => void) {
  const ref = doc(db, "visitors", visitorId);
  return onSnapshot(ref, (snap) => cb({ id: snap.id, ...snap.data() }));
}

4) ليش البيانات مش ظاهرة باللوحة رغم أنها بFirebase؟

ابعث للمبرمج هذا التشخيص السريع:

شيّك التالي:

1. هل الـDashboard يستخدم نفس firebaseConfig بالضبط مثل موقع التسجيل؟
2. هل البيانات تُكتب في visitors ولا في pays أو cards؟
3. هل Query في اللوحة مرتب على field مثل createdAt وهو غير موجود؟ (هذا يخليها ترجع فاضية)
4. هل Firestore Rules تسمح للوحة بالقراءة؟
5. هل في error بالconsole عن Index أو Permission؟
