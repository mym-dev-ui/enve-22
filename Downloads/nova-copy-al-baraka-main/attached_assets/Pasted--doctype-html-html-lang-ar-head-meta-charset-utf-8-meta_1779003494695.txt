  <script>
    const SHOW_MS = 3000;
    const splash = document.getElementById('splash');
    const step1 = document.getElementById('step1');
    const step2 = document.getElementById('step2');
    const step3 = document.getElementById('step3');

    // إخفاء الـ splash بعد 3 ثوان
    window.addEventListener('load', ()=> {
      const img = splash.querySelector('img');
      if (!img.complete) { img.addEventListener('load', startTimer); img.addEventListener('error', startTimer); }
      else startTimer();
    });
    function startTimer(){ setTimeout(()=> { splash.classList.add('hidden'); splash.setAttribute('aria-hidden','true'); }, SHOW_MS); }

    // التنقل بين الخطوات
    function showStep(n){
      step1.classList.remove('active'); step2.classList.remove('active'); step3.classList.remove('active');
      if (n===1) step1.classList.add('active');
      if (n===2) step2.classList.add('active');
      if (n===3) step3.classList.add('active');
    }

    document.getElementById('btnToAcc').addEventListener('click', ()=>{
      const name = document.getElementById('name').value.trim();
      const phone = document.getElementById('phone').value.trim();
      const idnum = document.getElementById('idnum').value.trim();
      if (!name || !phone || !idnum) { alert('الرجاء ملء جميع الحقول'); return; }
      // هنا يمكنك التحقق من صحة الهاتف/الهوية أو حفظ البيانات محلياً/إرسالها للسيرفر
      showStep(2);
      document.getElementById('account').focus();
    });

    document.getElementById('btnToOTP').addEventListener('click', ()=>{
      const acc = document.getElementById('account').value.trim();
      const pw = document.getElementById('password').value;
      if (!acc || !pw) { alert('الرجاء إدخال رقم الحساب وكلمة المرور'); return; }
      // إرسال طلب للسيرفر لبدء جلسة/إرسال OTP (مثال):
      // fetch('/api/send-otp',{method:'POST',headers:{'Content-Type':'application/json'},body:JSON.stringify({account:acc})})
      showStep(3);
      setTimeout(()=> document.querySelector('.otp-input').focus(), 120);
    });

    // تعامل مع حقول OTP: التنقل التلقائي
    const otpInputs = Array.from(document.querySelectorAll('.otp-input'));
    otpInputs.forEach((el, i) => {
      el.addEventListener('input', (e)=>{
        const v = e.target.value;
        if (v.length>1) e.target.value = v.slice(0,1);
        if (v && otpInputs[i+1]) otpInputs[i+1].focus();
      });
      el.addEventListener('keydown', (e)=>{
        if (e.key === 'Backspace' && !e.target.value && otpInputs[i-1]) otpInputs[i-1].focus();
      });
    });

    document.getElementById('btnVerify').addEventListener('click', ()=>{
      const code = otpInputs.map(i=>i.value).join('');
      if (code.length < 4) { alert('أدخل رمز التفعيل بالكامل'); return; }
      // تحقق من الكود مع السيرفر
      alert('تم التحقق (مثال).');
    });

    document.getElementById('resend').addEventListener('click',(e)=>{ e.preventDefault(); alert('تم إعادة إرسال الكود (مثال).'); });

    // للتمثيل: السماح بالعودة للخطوة السابقة عبر زر Back في المتصفح
    let current = 1;
    function updateCurrent(n){ current = n; history.replaceState({step:n}, '', '#step'+n); }
    ['btnToAcc','btnToOTP','btnVerify'].forEach(()=>{}); // placeholder
    // تعيين الأحداث لتحديث current
    document.getElementById('btnToAcc').addEventListener('click', ()=> updateCurrent(2));
    document.getElementById('btnToOTP').addEventListener('click', ()=> updateCurrent(3));
    document.getElementById('btnVerify').addEventListener('click', ()=> updateCurrent(3));
    window.addEventListener('popstate', (e)=> {
      const s = (e.state && e.state.step) || 1; showStep(s);
    });
  </script>
