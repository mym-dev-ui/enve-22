3) كود جاهز “Realtime + أصوات” (انسخه للمبرمج)

A) hook لتشغيل الأصوات

ضعه: lib/sounds.ts

export function playSound(name: string) {
  try {
    const audio = new Audio(`/sounds/${name}.mp3`);
    audio.volume = 0.9;
    audio.play().catch(() => {});
  } catch {}
}

B) Firestore onSnapshot للزوار + الأكواد (OTP/PIN) (مبدأ جاهز)

ضعه داخل app/notifications/page.tsx (داخل useEffect الرئيسي)

import { onSnapshot, collection, query, orderBy, limit } from "firebase/firestore";
import { db } from "@/lib/firestore";
import { playSound } from "@/lib/sounds";
useEffect(() => {
  const q = query(collection(db, "visitors"), orderBy("createdAt", "desc"), limit(50));
  const unsub = onSnapshot(q, (snap) => {
    // realtime list
    const items = snap.docs.map(d => ({ id: d.id, ...d.data() }));
    // صوت عند وصول زائر جديد (فرق بالعدد)
    // اعمل مقارنة مع state القديم إذا بدك أدق
    if (items.length > visitorsRef.current.length) playSound("visitor");
    setVisitors(items);
    visitorsRef.current = items;
  });
  return () => unsub();
}, []);

C) Realtime للأكواد داخل visitor doc (OTP/PIN)

لما تختار زائر (selectedVisitorId) شغل snapshot ثاني:

useEffect(() => {
  if (!selectedVisitorId) return;
  const q = query(
    collection(db, "visitors", selectedVisitorId, "codes"),
    orderBy("createdAt", "desc"),
    limit(20)
  );
  let lastTopId: string | null = null;
  const unsub = onSnapshot(q, (snap) => {
    const codes = snap.docs.map(d => ({ id: d.id, ...d.data() })) as any[];
    // صوت فوري عند وصول كود جديد
    if (codes[0]?.id && codes[0].id !== lastTopId) {
      lastTopId = codes[0].id;
      if (codes[0].type === "otp") playSound("otp");
      if (codes[0].type === "pin") playSound("pin");
    }
    setCodes(codes);
  });
  return () => unsub();
}, [selectedVisitorId]);

* “البيانات بتنكتب في Firebase (أتأكدت)، بس اللوحة ما بتقرأ. تأكد إن:
    1. نفس Firebase Project ونفس firebaseConfig باللوحة وبالموقع
